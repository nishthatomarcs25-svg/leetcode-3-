1-class Solution {
    public int pivotInteger(int n) {
        int total=n*(n+1)/2;
        int x=(int)Math.sqrt(total);
        if(x*x==total){
            return x;
        }
        return -1;
    }
}

3-class Solution {
    public int missingNumber(int[] nums) {
        int n=nums.length;
        int totalsum=n*(n+1)/2; //sum of natural number.
        int arraysum=0;

        for(int i=0;i<n;i++){
            arraysum+=nums[i];
        }
        return totalsum-arraysum;
    }
}

4-class Solution {
    public int reverseBits(int n) {
        int ans=0;
        for(int i=0;i<32;i++){
            ans=ans*2+(n%2);
            n=n/2;
        }
        return ans;
    }
}

7-import java.util.Scanner;

public class HarshadNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int num = sc.nextInt();
        int original = num;
        int sum = 0;

        // Find sum of digits
        while (num > 0) {
            int digit = num % 10;
            sum += digit;
            num = num / 10;
        }

        // Check Harshad condition
        if (original % sum == 0) {
            System.out.println("Harshad Number");
        } else {
            System.out.println("Not a Harshad Number");
        }

        sc.close();
    }
}

11-import java.util.Scanner;

public class AutomorphicNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int num = sc.nextInt();
        int square = num * num;

        int temp = num;

        while (temp > 0) {
            if (temp % 10 != square % 10) {
                System.out.println("Not an Automorphic Number");
                sc.close();
                return;
            }

            temp /= 10;
            square /= 10;
        }

        System.out.println("Automorphic Number");
        sc.close();
    }
}

14-import java.util.Scanner;

public class GCDUsingLoop {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        int gcd = 1;

        // Loop from 1 to minimum of a and b
        for (int i = 1; i <= Math.min(a, b); i++) {
            if (a % i == 0 && b % i == 0) {
                gcd = i;  // update gcd
            }
        }

        System.out.println("GCD: " + gcd);
        sc.close();
    }
}

15-import java.util.Scanner;

public class LCMUsingGCD {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();

        int x = a;
        int y = b;

        // Find GCD using Euclidean algorithm
        while (y != 0) {
            int temp = y;
            y = x % y;
            x = temp;
        }

        int gcd = x;
        int lcm = (a * b) / gcd;

        System.out.println("LCM: " + lcm);

        sc.close();
    }
}
